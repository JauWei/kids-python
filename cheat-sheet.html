🔤 字串 Strings

s = "Hello Python"
len(s)               # 12
s.upper()            # HELLO PYTHON
s.lower()            # hello python
s.replace("H", "J")  # Jello Python
s.strip()            # 去掉前後空白

"Py" in s          # True
"-" * 10            # "----------"

✨ f-string Formatted

字串前加 f,大括號裡可放變數或算式。

name = "小華"
age = 13
print(f"我叫 {name}, {age} 歲")
print(f"明年 {age + 1} 歲")
print(f"{3.14159:.2f}")  # 3.14 (小數兩位)

height = float(input("身高 m？ "))

🔁 for + range

for i in range(5):
    print(i)         # 0,1,2,3,4 (5 個)

range(2, 8)         # 2..7
range(0, 10, 2)     # 0,2,4,6,8
range(10, 0, -1)    # 10..1 倒數

list(range(5))      # 攤開看

range(N) = N 個數字,從 0 開始 ── 不是「到 N」。

♾️ while

📋 list

nums = [10, 20, 30, 40]
nums[0]             # 10  (從 0 開始)
nums[-1]            # 40  (尾巴)
nums[1:3]           # [20,30]  切片
len(nums)           # 4

nums.append(50)     # 加在尾
nums.remove(20)     # 移除值
nums.insert(1, 99)  # 插入到位置
30 in nums          # True

for n in nums:
    print(n)

📓 dict

for k in d:
    print(k, d[k])

🛠️ 函式 def

def add(a, b):
    return a + b

def greet(name, msg="嗨"):    # 預設值
    print(f"{msg}, {name}!")

result = add(3, 5)         # 8
greet("小明")              # 嗨, 小明!
greet("Alex", "Hello")     # Hello, Alex!

return 把值傳回去;print 只是顯示。

🛡️ try / except

try:
    n = int(input("數字： "))
    print(10 / n)
except ValueError:
    print("那不是數字")
except ZeroDivisionError:
    print("不能除以 0")
except:
    print("其他錯誤")

常見錯誤:ValueError / TypeError / IndexError / KeyError / ZeroDivisionError。

🧬 class

class Dog:
    def __init__(self, name, breed):
        self.name = name
        self.breed = breed

    def bark(self):
        print(f"{self.name} 汪!")

d = Dog("旺旺", "柴犬")
d.bark()
print(d.name)

# 繼承
class Puppy(Dog):
    def play(self):
        print("玩球")

📁 檔案 I/O

# 寫入(會覆蓋)
with open("a.txt", "w") as f:
    f.write("hello\n")

# 讀取
with open("a.txt", "r") as f:
    content = f.read()

# 一行一行
with open("a.txt") as f:
    for line in f:
        print(line.strip())

# 追加
with open("a.txt", "a") as f:
    f.write("more\n")
